#define LED_PIN 3
#define LDR_PIN A2
#define THRESHOLD 100
#define PERIOD 15
const int LED=13;
const int LED2=11;

#include <LiquidCrystal_I2C.h> //DESCARGAR LIBRERÃA:https://github.com/ELECTROALL/Codigos-arduino/blob/master/LiquidCrystal_I2C.zip
LiquidCrystal_I2C lcd(0x27,16,2); // si no te sale con esta direccion  puedes usar (0x3f,16,2) || (0x27,16,2)  ||(0x20,16,2) 

bool previous_state;
bool current_state;

void setup() 
{
    Serial.begin(9600);
    pinMode(LED_PIN, OUTPUT);
  lcd.init();
  lcd.backlight();
  lcd.clear();
  //lcd.setCursor(0,0);
  lcd.print("==> TEXTO <=="); 
 delay(2000);
   lcd.clear();
  //lcd.setCursor (0,1);
  //lcd.print("==> ELECTROMARK <");

  pinMode(LED,OUTPUT);
  pinMode(LED2,OUTPUT);
  
}

void loop() 
{
  current_state = get_ldr();
  if(!current_state && previous_state)
  {
    print_byte(get_byte());
  }
  previous_state = current_state;
}

bool get_ldr()
{
  int voltage = analogRead(LDR_PIN);
  return voltage > THRESHOLD ? true : false;
   

}

char get_byte()
{
  char ret = 0;
  delay(PERIOD*1.5);
  for(int i = 0; i < 8; i++)
  {
   ret = ret | get_ldr() << i; 
   delay(PERIOD);
   
 
    if (ret==0)
    {
      digitalWrite(LED,LOW);
       digitalWrite(LED2,HIGH);
     // lcd.clear();
   }else
   {
        
       digitalWrite(LED2,LOW); 
    //    delay(100);  
     Serial.print(ret);
   }
   
  }
  return ret;
}
void print_byte(char my_byte)
{
  char buff[2];
  sprintf(buff, "%c", my_byte);
  Serial.print(buff);
  //lcd.setCursor(0,0);
  lcd.print(buff);
 //    digitalWrite(LED,HIGH); 

  
}

#define LED_PIN A1
#define BUTTON_PIN A0
#define PERIOD 15


char* string = " HOLA MUNDO ";
int string_length;

void setup()
{
  pinMode(LED_PIN, OUTPUT);
  //pinMode(BUTTON_PIN, INPUT_PULLUP);
  string_length = strlen(string);
}

void loop()
{
  for(int i = 0; i < string_length; i ++)
  {
    send_byte(string[i]);
  }
  delay(100);
}

void send_byte(char my_byte)
{
  digitalWrite(LED_PIN, LOW);
  delay(PERIOD);

  //transmission of bits
  for(int i = 0; i < 8; i++)
  {
    digitalWrite(LED_PIN, (my_byte&(0x01 << i))!=0 );
    delay(PERIOD);
  }

  digitalWrite(LED_PIN, HIGH);
  delay(PERIOD);

}
